SQL:

WITH cte AS (
    SELECT
        D.name AS Department,
        E.name AS Employee,
        E.salary AS Salary,
        DENSE_RANK() OVER (
            PARTITION BY D.name
            ORDER BY E.salary DESC
        ) AS rnk
    FROM Employee E
    JOIN Department D
      ON E.departmentId = D.id
)
SELECT
    Department,
    Employee,
    Salary
FROM cte
WHERE rnk = 1;
